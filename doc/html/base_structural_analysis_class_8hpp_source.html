483 ::computeDegreesOfFreedom()
484 {
485  unsigned int numNode = m_nodeList.size();
486  unsigned int numNodeDegFree = m_maxNodeDegFree;
487  unsigned int numDof = numNode*numNodeDegFree;
488 
489  if( m_isInitialStructureSaved && (!m_isDegreeOfFreedomVecComputed) )
490  {
491  Eigen::Matrix<unsigned int,Eigen::Dynamic,1> degreesFreedomVec = Eigen::Matrix<unsigned int, Eigen::Dynamic, 1>::Zero(numDof);
492  Eigen::Matrix<numericType,Eigen::Dynamic,1> curNode_isFixedVec;
493  unsigned int actualDegreeFreedom = 0;
494  unsigned int curNode;
495  unsigned int firstNonFixed = 0;
496  unsigned int nextFixed = 0;
497  unsigned int curDegAtNode;
498 
499  // Searching all nodes
500  for (auto it = m_fixedNodes.begin(); it != m_fixedNodes.end(); ++it) {
501  nextFixed = *it;
502  for (curNode = firstNonFixed; curNode < nextFixed; ++curNode) {
503  // Searching all "DoF" at current node
504  for (curDegAtNode = 0; curDegAtNode < numNodeDegFree; ++curDegAtNode) {
505  // If it is an actual free degree, update the number of actual DoF and save it
506  degreesFreedomVec(curDegAtNode + curNode * numNodeDegFree) = ++actualDegreeFreedom;
507  }
508  }
509  m_nTotalDoF = actualDegreeFreedom;
510  m_degFreeVec = degreesFreedomVec;
511 
512  firstNonFixed = nextFixed+1;
513  }
514  }
515 
516  // Updating flag
517  m_isDegreeOfFreedomVecComputed = (m_isInitialStructureSaved);
518 
519  // Returning suitable value
520  if(m_isDegreeOfFreedomVecComputed) { return m_degFreeVec; }
521  else return Eigen::Matrix<unsigned int, Eigen::Dynamic, 1>::Zero(numDof);
522 }

524 template<class numericType, class elemClass, class nodeClass>
525 Eigen::SparseMatrix<numericType> baseStructuralAnalysisClass<numericType,elemClass,nodeClass>
526 ::evaluateGlobalStiffMat()
527 {
528  // Getting constantes and some auxiliar variables
529  unsigned int numNodePerElem = m_nNodePerElem; // Number of nodes per element [USING =2]!!!!!!!!!!!!!
530  unsigned int numNodeDegFree = m_maxNodeDegFree; // Number of possible degrees of freedom in each node
531  unsigned int numElemDegFree = numNodePerElem*numNodeDegFree;// Number of possible degrees of freedom in each elem
532  unsigned int numElem = m_elemList.size(); // Number of elements
533  unsigned int numNode = m_nodeList.size(); // Number of nodes
534  unsigned int idCurElem; // Index of an element
535  unsigned int idNode_curElem; // Initial node of an element
536 
537  // Global Stiffness Matrix
538  Eigen::SparseMatrix<numericType> glbStiffMatrix(numNode*numNodeDegFree,numNode*numNodeDegFree);
539 
540  if( (m_isInitialStructureSaved && m_isDegreeOfFreedomVecComputed ) && (!m_isGlbStiffMatComputed) )
541  {
542  // Getting some small vectors and matrix
543  Eigen::Matrix<unsigned int,Eigen::Dynamic,1> elemDofPosition_curElem(numElemDegFree);
544  Eigen::Matrix<unsigned int,Eigen::Dynamic,1> iniNodeDofPosition_curElem(numNodeDegFree);
545  Eigen::Matrix<unsigned int,Eigen::Dynamic,1> endNodeDofPosition_curElem(numNodeDegFree);
546  std::vector<unsigned int> elemConnection_curElem(numNodePerElem); // ConnectionVector of an element
547  Eigen::SparseMatrix<numericType> localStifffMat_CurElem(numElemDegFree,numElemDegFree); // localStiffMatrix of an element
548  Eigen::Matrix<numericType,Eigen::Dynamic,Eigen::Dynamic> rotationMat_CurElem(numElemDegFree,numElemDegFree); // rotationMatrix of an element
549  typedef Eigen::Triplet<numericType> TripNumericType; // Triplet typedef for a sparse matrix
550  std::vector<TripNumericType> tripletList; // Triplet list for indexzation of sparse matrix
551  tripletList.reserve(numElemDegFree*numElemDegFree*numElem); // Allocating memory for the triplet list
552  unsigned int curRow_LclStfMat =0; // Index of a row at localStiffMatrix
553  unsigned int curCol_LclStfMat =0; // Index of a col at localStiffMatrix
554  // unsigned int curRowDof_LclStfMat =0; // DoF of the IndexRow @ localStiffMatrix
555  // unsigned int curColDof_LclStfMat =0; // DoF of the IndexCol @ localStiffMatrix
556  unsigned int glbIdRow = 0; // Index of the row @ globalStiffMatrix
557  unsigned int glbIdCol = 0; // Index of the col @ globalStiffMatrix
558  numericType stiff; // An element of the localStiffMatrix
559  numericType maxValue = 0; // Variable to compute the max value of localStiffMatrix
560 
561  // Trick...
562  // Walking inside the elementList: elem by elem
563  for( idCurElem=0; idCurElem<numElem; ++idCurElem)
564  {
565  // Saving the connection (id nodes) of an element
566  elemConnection_curElem = m_elemList[idCurElem].getM_nodalConnectivity();
567 
568  // Getting the 'position' of the DoF of the nodes that connects an element
569  for(idNode_curElem = 0; idNode_curElem < numNodePerElem; ++idNode_curElem){
570  elemDofPosition_curElem.block(idNode_curElem*numNodeDegFree,0,numNodeDegFree,1) = Matrix<unsigned int,Eigen::Dynamic,1>::LinSpaced(numNodeDegFree,elemConnection_curElem[idNode_curElem]*numNodeDegFree,(elemConnection_curElem[idNode_curElem]+1u)*numNodeDegFree - 1u);
571  }
572 
573  // Computing the local stiffness matrix of an element and its rotation matrix
574  localStifffMat_CurElem = m_elemList[idCurElem].sprMatK(m_youngModulus, m_poissonRatio);
575 
576  // Walking inside the localStiffMatrix: row by row
577  for( curRow_LclStfMat = 0; curRow_LclStfMat<numElemDegFree ; ++curRow_LclStfMat)
578  {
579  glbIdRow = elemDofPosition_curElem(curRow_LclStfMat);
580 
581  // Walking inside the localStiffMatrix: col by col
582  for( curCol_LclStfMat = 0; curCol_LclStfMat<numElemDegFree ; ++curCol_LclStfMat)
583  {
584  glbIdCol = elemDofPosition_curElem(curCol_LclStfMat);
585 
586  // Getting the stiffness of localStiffMatrix roxXcol and building the triplet
587  stiff = localStifffMat_CurElem.coeffRef(curRow_LclStfMat, curCol_LclStfMat);
588  tripletList.push_back( TripNumericType( glbIdRow , glbIdCol , stiff ) );
589 
590  // Computing the maxValue of diag(globalStiffMatrix)
591  if( (glbIdRow == glbIdCol) && (maxValue < stiff) ) maxValue = stiff;
592  }
593  }
594  }
595 
596  // Saving global stiffness matrix as sparse (from triplets(i,j,v_ij))
597  glbStiffMatrix.setFromTriplets(tripletList.begin(), tripletList.end());
598  m_glbStiffMat = glbStiffMatrix;
599 
600  // Saving penalty value at class member m_penalty
601  m_penalty = maxValue * m_penalty_multiplier;
602  }
603 
604  // Updating flag
605  m_isGlbStiffMatComputed = (m_isInitialStructureSaved && m_isDegreeOfFreedomVecComputed );
606 
607  // Returning suitable value
608  if(m_isGlbStiffMatComputed) { return m_glbStiffMat; }
609  else return glbStiffMatrix;
610 }
