439 Eigen::Matrix<unsigned int, Eigen::Dynamic, 1> baseStructuralAnalysisClass<numericType,elemClass,nodeClass>
440 ::computeDegreesOfFreedom()
441 {
442  unsigned int numNode = m_nodeList.size();
443  unsigned int numNodeDegFree = m_maxNodeDegFree;
444  unsigned int numDof = numNode*numNodeDegFree;
445 
446  if( m_isInitialStructureSaved && (!m_isDegreeOfFreedomVecComputed) )
447  {
448  Eigen::Matrix<unsigned int,Eigen::Dynamic,1> degreesFreedomVec = Eigen::Matrix<unsigned int, Eigen::Dynamic, 1>::Zero(numDof);
449  Eigen::Matrix<numericType,Eigen::Dynamic,1> curNode_isFixedVec;
450  unsigned int actualDegreeFreedom = 0;
451  unsigned int curNode;
452  unsigned int curDegAtNode;
453 
454  // Searching all nodes
455  for(curNode = 0; curNode<numNode; ++curNode)
456  {
457  // Saving "DoF" of each node
458  curNode_isFixedVec = m_nodeList[curNode].isItFixed();
459 
460  // Searching all "DoF" at current node
461  for(curDegAtNode = 0; curDegAtNode<numNodeDegFree; ++curDegAtNode)
462  {
463  // Verifying each "DoF" at current node
464  if( curNode_isFixedVec(curDegAtNode) == 0 )
465  {
466  // If it is an actual free degree, update the number of actual DoF and save it
467  degreesFreedomVec(curNode*numNodeDegFree+curDegAtNode) = ++actualDegreeFreedom;
468  }
469  }
470  }
471  m_nTotalDoF = actualDegreeFreedom;
472  m_degFreeVec = degreesFreedomVec;
473  }
474 
475  // Updating flag
476  m_isDegreeOfFreedomVecComputed = (m_isInitialStructureSaved);
477 
478  // Returning suitable value
479  if(m_isDegreeOfFreedomVecComputed) { return m_degFreeVec; }
480  else return Eigen::Matrix<unsigned int, Eigen::Dynamic, 1>::Zero(numDof);
481 }

483 template<class numericType, class elemClass, class nodeClass>

485 ::evaluateGlobalStiffMat()
486 {
487  // Getting constantes and some auxiliar variables
488  unsigned int numNodePerElem = m_nNodePerElem; // Number of nodes per element [USING =2]!!!!!!!!!!!!!
489  unsigned int numNodeDegFree = m_maxNodeDegFree; // Number of possible degrees of freedom in each node
490  unsigned int numElemDegFree = numNodePerElem*numNodeDegFree;// Number of possible degrees of freedom in each elem
491  unsigned int numElem = m_elemList.size(); // Number of elements
492  unsigned int numNode = m_nodeList.size(); // Number of nodes
493  unsigned int idCurElem; // Index of an element
494  unsigned int iniNode_CurElem; // Initial node of an element
495  unsigned int endNode_CurElem; // End node of an element
496 
497  // Global Stiffness Matrix
498  Eigen::SparseMatrix<numericType> glbStiffMatrix(numNode,numNode);
499 
500  if( (m_isInitialStructureSaved && m_isDegreeOfFreedomVecComputed && m_isLocalRotationMatrixComputed) && (!m_isGlbStiffMatComputed) )
501  {
502  // Getting some small vectors and matrix
503  Eigen::Matrix<unsigned int,Eigen::Dynamic,1> elemDofPosition_curElem(numElemDegFree);
504  Eigen::Matrix<unsigned int,Eigen::Dynamic,1> iniNodeDofPosition_curElem(numNodeDegFree);
505  Eigen::Matrix<unsigned int,Eigen::Dynamic,1> endNodeDofPosition_curElem(numNodeDegFree);
506  Eigen::Matrix<unsigned int,Eigen::Dynamic,1> elemConnection_curElem(numNodePerElem); // ConnectionVector of an element
507  Eigen::Matrix<numericType,Eigen::Dynamic,Eigen::Dynamic> localStifffMat_CurElem(numElemDegFree,numElemDegFree); // localStiffMatrix of an element
508  Eigen::Matrix<numericType,Eigen::Dynamic,Eigen::Dynamic> rotationMat_CurElem(numElemDegFree,numElemDegFree); // rotationMatrix of an element
509  typedef Eigen::Triplet<numericType> TripNumericType; // Triplet typedef for a sparse matrix
510  std::vector<TripNumericType> tripletList; // Triplet list for indexzation of sparse matrix
511  tripletList.reserve(numElemDegFree*numElemDegFree*numElem); // Allocating memory for the triplet list
512  unsigned int curRow_LclStfMat =0; // Index of a row at localStiffMatrix
513  unsigned int curCol_LclStfMat =0; // Index of a col at localStiffMatrix
514  // unsigned int curRowDof_LclStfMat =0; // DoF of the IndexRow @ localStiffMatrix
515  // unsigned int curColDof_LclStfMat =0; // DoF of the IndexCol @ localStiffMatrix
516  unsigned int glbIdRow = 0; // Index of the row @ globalStiffMatrix
517  unsigned int glbIdCol = 0; // Index of the col @ globalStiffMatrix
518  numericType stiff; // An element of the localStiffMatrix
519  numericType maxValue = 0; // Variable to compute the max value of localStiffMatrix
520 
521  // Trick...
522  // Walking inside the elementList: elem by elem
523  for( idCurElem=0; idCurElem<numElem; ++idCurElem)
524  {
525  // Saving the connection (id nodes) of an element
526  elemConnection_curElem = m_elemList[idCurElem].getNodeConnection();
527  iniNode_CurElem = elemConnection_curElem(0);
528  endNode_CurElem = elemConnection_curElem(1);
529 
530  // // Getting the DoFs of the nodes that connects an element
531  // Eigen::Matrix<unsigned int,Eigen::Dynamic,1> elemDoF_curElem(numRowLclStiff);
532  // elemDoF_curElem << m_degFreeVec.segment(iniNode_CurElem*numNodeDegFree,numNodeDegFree);
533  // elemDoF_curElem << m_degFreeVec.segment(endNode_CurElem*numNodeDegFree,numNodeDegFree);
534 
535  // Getting the 'position' of the DoF of the nodes that connects an element
536  iniNodeDofPosition_curElem = Matrix<unsigned int,Eigen::Dynamic,1>::LinSpaced(numNodeDegFree,iniNode_CurElem*numNodeDegFree,(iniNode_CurElem+1u)*numNodeDegFree - 1u);
537  endNodeDofPosition_curElem = Matrix<unsigned int,Eigen::Dynamic,1>::LinSpaced(numNodeDegFree,endNode_CurElem*numNodeDegFree,(endNode_CurElem+1u)*numNodeDegFree - 1u);
538  elemDofPosition_curElem << iniNodeDofPosition_curElem,endNodeDofPosition_curElem;
539 
540  // Computing the local stiffness matrix of an element and its rotation matrix
541  localStifffMat_CurElem = m_elemList[idCurElem].localStiffMatrix();
542  rotationMat_CurElem = m_elemList[idCurElem].getRotMat();
543 
544  // Applying rotation to get it in global coordinates
545  localStifffMat_CurElem = rotationMat_CurElem.transpose() * localStifffMat_CurElem * rotationMat_CurElem;
546 
547  // Walking inside the localStiffMatrix: row by row
548  for( curRow_LclStfMat = 0; curRow_LclStfMat<numElemDegFree ; ++curRow_LclStfMat)
549  {
550  glbIdRow = elemDofPosition_curElem(curRow_LclStfMat);
551  // // Get the DoF of this specific row (corresponding DoF of the node)
552  // curRowDof_LclStfMat = elemDoF_curElem(curRow_LclStfMat);
553  // if( curRowDof_LclStfMat = 0 ) { continue; } // Neglet this row if DoF=0 (its not free!)
554 
555  // Walking inside the localStiffMatrix: col by col
556  for( curCol_LclStfMat = 0; curCol_LclStfMat<numElemDegFree ; ++curCol_LclStfMat)
557  {
558  glbIdCol = elemDofPosition_curElem(curCol_LclStfMat);
559  // // Get the DoF of this specific col (corresponding DoF of the node)
560  // curColDof_LclStfMat = elemDoF_curElem(curCol_LclStfMat);
561  // if( curColDof_LclStfMat = 0 ) { continue; } // Neglet this col if DoF=0 (its not free!)
562 
563  // Getting the stiffness of localStiffMatrix roxXcol and building the triplet
564  stiff = localStifffMat_CurElem(curRow_LclStfMat, curCol_LclStfMat);
565  tripletList.push_back( TripNumericType( glbIdRow , glbIdCol , stiff ) );
566 
567  // Computing the maxValue of diag(globalStiffMatrix)
568  if( (glbIdRow == glbIdCol) && (maxValue < stiff) ) maxValue = stiff;
569  }
570  }
571  }
572 
573  // Saving global stiffness matrix as sparse (from triplets(i,j,v_ij))
574  glbStiffMatrix.setFromTriplets(tripletList.begin(), tripletList.end());
575  m_glbStiffMat = glbStiffMatrix;
576 
577  // Saving penalty value at class member m_penalty
578  m_penalty = maxValue * m_penalty_multiplier;
579  }
580 
581  // Updating flag
582  m_isGlbStiffMatComputed = (m_isInitialStructureSaved && m_isDegreeOfFreedomVecComputed && m_isLocalRotationMatrixComputed);
583 
584  // Returning suitable value
585  if(m_isGlbStiffMatComputed) { return m_glbStiffMat; }
586  else return glbStiffMatrix;
587 }
