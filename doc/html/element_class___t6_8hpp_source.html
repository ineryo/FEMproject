289 Eigen::SparseMatrix<numericType> elementClass_T6<numericType>::
290 evalConstitutiveMatrix() const{
291 
292  // Auxiliar objects to help fill a sparse matrix
293  typedef Eigen::Triplet<numericType> TripNumericType; // Triplet typedef for a sparse matrix
294  std::vector<TripNumericType> tripletList; // Triplet list for indexation of sparse matrix
295  tripletList.reserve(5); // Allocating memory for the triplet list
296 
297  Eigen::SparseMatrix<numericType> C(3,3); // Sparse matrix to be returned (must be filled)
298 
299  // Plane Stress:
300  numericType poisson = this->m_poissonRatio; // Poisson
301  numericType auxValue1 = this->m_youngModulus/(1.-poisson*poisson); // auxValue1 = E/(1-v^2)
302  numericType auxValue2 = poisson*auxValue1; // auxValue2 = v*E/(1-v^2)
303 
304 
305  // Feeding triplet to fill the sparse matrix
306  tripletList.push_back(TripNumericType(0, 0, auxValue1));
307  tripletList.push_back(TripNumericType(0, 1, auxValue2));
308  tripletList.push_back(TripNumericType(1, 0, auxValue2));
309  tripletList.push_back(TripNumericType(1, 1, auxValue1));
310  tripletList.push_back(TripNumericType(2, 2, auxValue1*(1.-poisson)*.5));
311 
312  C.setFromTriplets(tripletList.begin(), tripletList.end()); // Filling the sparse matrix
313  return C; // Returning the answer
